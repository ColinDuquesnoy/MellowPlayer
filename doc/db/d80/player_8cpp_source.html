228 void PlayerInterface::checkPlaybackStatusChange(SongInfo &song)
229 {
230  PlaybackStatus status = Stopped;
231  if(song.isValid())
232  status = playbackStatus();
233  if(status != m_currentStatus)

237  m_currentStatus = status;

242 //---------------------------------------------------------
243 void PlayerInterface::checkSongChange(SongInfo &song)
244 {
245  if(song.songName != m_currentSong.songName)
246  {
247  qDebug() << "Song changed: " << song.songName;
248  emit songChanged(song);
249  m_currentSong = song;
250 
251  if(playbackStatus() == Playing && song.isValid())
252  {
253  m_startedSong = song;
254  emit songStarted(song);
255  }
256  else
257  m_startedSong = SongInfo(); // invalid song
258 
259  if(song.isValid())
260  downloadSongArt(song.artUrl);
261  }
262  else if(!m_startedSong.isValid() && playbackStatus() == Playing)
263  {
264  // sometimes (depends on the player) a new song is active but not playing,
265  // we only emit song started when a song is active and the player is playing.
266  m_startedSong = song;
267  emit songStarted(song);
268  }
269 }

282 void PlayerInterface::checkForControlCaps()
283 {
284  IStreamingService* iService =
285  Services::streamingServices()->currentService();
286  if(iService)
287  {
288  if(canGoNext() != m_canGoNext ||
289  canGoPrevious() != m_canGoPrevious ||
290  canPlay() != m_canPlay ||
291  canSeek() != m_canSeek)
292  {
293  m_canGoNext = canGoNext();
294  m_canGoPrevious = canGoPrevious();
295  m_canPlay = canPlay();
296  m_canSeek = canSeek();
297 
298  emit controlCapsChanged();
299  }
300  }
301 }

305 {
306  UrlDownloader* downloader = new UrlDownloader(this);
307  QString dest = QDir::temp().path() +
308  QString(QDir::separator()) +
309  currentSong().songId +
310  QFileInfo(url).fileName();
311  connect(downloader, SIGNAL(finished(const QString&)),
312  this, SLOT(onArtReady(const QString&)));
313 
314  downloader->download(url, dest);
